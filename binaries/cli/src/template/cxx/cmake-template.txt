cmake_minimum_required(VERSION 3.21)
project(cxx-dataflow LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_FLAGS "-fPIC")

set(ADORA_ROOT_DIR "__ADORA_PATH__" CACHE FILEPATH "Path to the root of adora")

set(adora_cxx_include_dir "${CMAKE_CURRENT_BINARY_DIR}/include/cxx")
set(node_bridge "${CMAKE_CURRENT_BINARY_DIR}/node_bridge.cc")

if(ADORA_ROOT_DIR)
    include(ExternalProject)
    ExternalProject_Add(Adora
        SOURCE_DIR ${ADORA_ROOT_DIR}
        BUILD_IN_SOURCE True
        CONFIGURE_COMMAND ""
        BUILD_COMMAND
            cargo build
            --package adora-node-api-cxx
        INSTALL_COMMAND ""
    )

    add_custom_command(OUTPUT ${node_bridge} ${adora_cxx_include_dir} ${operator_bridge} ${adora_c_include_dir}
        WORKING_DIRECTORY ${ADORA_ROOT_DIR}
        DEPENDS Adora
        COMMAND
            mkdir ${adora_cxx_include_dir} -p
            &&
            cp target/cxxbridge/adora-node-api-cxx/src/lib.rs.cc ${node_bridge}
            &&
            cp target/cxxbridge/adora-node-api-cxx/src/lib.rs.h ${adora_cxx_include_dir}/adora-node-api.h
    )
    
    add_custom_target(Adora_cxx DEPENDS ${node_bridge} ${adora_cxx_include_dir})
    set(adora_link_dirs ${ADORA_ROOT_DIR}/target/debug)
else()
    include(ExternalProject)
    ExternalProject_Add(Adora
        PREFIX ${CMAKE_CURRENT_BINARY_DIR}/adora
        GIT_REPOSITORY https://github.com/adora-rs/adora.git
        GIT_TAG main
        BUILD_IN_SOURCE True
        CONFIGURE_COMMAND ""
        BUILD_COMMAND
            cargo build
            --package adora-node-api-cxx
            --target-dir ${CMAKE_CURRENT_BINARY_DIR}/adora/src/Adora/target
        INSTALL_COMMAND ""
    )

    add_custom_command(OUTPUT ${node_bridge} ${adora_cxx_include_dir}
        WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/adora/src/Adora/target
        DEPENDS Adora
        COMMAND
            mkdir ${adora_cxx_include_dir} -p
            &&
            cp cxxbridge/adora-node-api-cxx/src/lib.rs.cc ${node_bridge}
            &&
            cp cxxbridge/adora-node-api-cxx/src/lib.rs.h ${adora_cxx_include_dir}/adora-node-api.h
    )

    set(adora_link_dirs ${CMAKE_CURRENT_BINARY_DIR}/adora/src/Adora/target/debug)

    add_custom_target(Adora_cxx DEPENDS ${node_bridge} ${adora_cxx_include_dir})
endif()

link_directories(${adora_link_dirs})

add_executable(talker_1 talker_1/node.cc ${node_bridge})
add_dependencies(talker_1 Adora_cxx)
target_include_directories(talker_1 PRIVATE ${adora_cxx_include_dir})
target_link_libraries(talker_1 adora_node_api_cxx z)

add_executable(talker_2 talker_2/node.cc ${node_bridge})
add_dependencies(talker_2 Adora_cxx)
target_include_directories(talker_2 PRIVATE ${adora_cxx_include_dir})
target_link_libraries(talker_2 adora_node_api_cxx z)

add_executable(listener_1 listener_1/node.cc ${node_bridge})
add_dependencies(listener_1 Adora_cxx)
target_include_directories(listener_1 PRIVATE ${adora_cxx_include_dir})
target_link_libraries(listener_1 adora_node_api_cxx z)

install(TARGETS listener_1 talker_1 talker_2 DESTINATION ${CMAKE_CURRENT_SOURCE_DIR}/bin)
